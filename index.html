async function optimizeDispatch(){ const mode = document.getElementById('dispatch-mode').value; // 체크된 + 보이는 업체만 선택 const selected = todayItems.filter(v => { const cb = document.getElementById('vc-' + v.id); if (!cb || !cb.checked) return false; const item = cb.closest('.vendor-item'); return item && !item.classList.contains('hidden-region'); }); if (selected.length < 1) { alert('업체를 선택해주세요'); return; } if (selected.length === 1) { dispatchResult = [selected]; renderDispatchResult([selected]); return; } let groups = []; if (mode === 'region') { // ── 배차지역 기반 그룹핑 ── const regionMap = {}; selected.forEach(v => { const r = v.region || '미지정'; if (!regionMap[r]) regionMap[r] = []; regionMap[r].push(v); }); groups = Object.values(regionMap); } else { // ── 거리 기반 그룹핑 (서울·대전은 동일 지역끼리만, 나머지는 거리 기반) ── // 좌표 없는 업체 경고 const noCoords = selected.filter(v => !v.lat || v.lat === ''); if (noCoords.length) { alert(`좌표가 없는 업체:\n${noCoords.map(v => v.name).join('\n')}\n\n구글 시트 Cognex_Vendors에서 lat/lng를 입력해주세요.`); return; } // 지역 분류: 서울·대전은 별도 풀로 분리, 나머지는 일반 풀 const FIXED_REGIONS = ['서울', '대전']; // 추후 추가 가능 const pools = {}; // { '서울': [...], '대전': [...], '__other': [...] } selected.forEach(v => { const r = (v.region || '').trim(); const key = FIXED_REGIONS.includes(r) ? r : '__other'; if (!pools[key]) pools[key] = []; pools[key].push(v); }); // 거리 기반 그룹핑 함수 (한 풀 안에서만 동작) function groupByDistance(pool, maxDist = 20, maxPerVehicle = 3) { const result = []; const remaining = [...pool]; while (remaining.length > 0) { const base = remaining.shift(); const group = [base]; // 가장 가까운 업체부터 채우기 let changed = true; while (changed && group.length < maxPerVehicle) { changed = false; let bestIdx = -1, bestDist = maxDist; remaining.forEach((v, i) => { // 그룹 내 모든 업체와의 최대 거리가 maxDist 이내인지 확인 const maxD = Math.max(...group.map(g => dist(g, v))); if (maxD < bestDist) { bestDist = maxD; bestIdx = i; } }); if (bestIdx >= 0) { group.push(remaining.splice(bestIdx, 1)[0]); changed = true; } } result.push(group); } return result; } // 서울·대전: 동일 지역 내에서만 거리 기반 분리 FIXED_REGIONS.forEach(r => { if (pools[r] && pools[r].length > 0) { groupByDistance(pools[r], 15, 4).forEach(g => groups.push(g)); } }); // 나머지: 기존 거리 기반 그룹핑 if (pools['__other'] && pools['__other'].length > 0) { groupByDistance(pools['__other'], 20, 3).forEach(g => groups.push(g)); } } dispatchResult = groups; renderDispatchResult(groups); }
